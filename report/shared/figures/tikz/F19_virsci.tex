  \node[state, below left=1.0cm and 0.7cm of au1]  (au2) {};
  \node[state, below right=1.0cm and 0.7cm of au1] (au3) {};
  \node[band label, below=0.45cm of au2, anchor=north] (cog) {\figlabel{F19.coauthorg}};
  % undirected co-author edges
  \draw[thick, draw=ari-blue] (au1) -- (au2);
