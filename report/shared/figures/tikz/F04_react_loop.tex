% F04: per-node ReAct driver loop — Think / Act / Observe / Reflect as a
%      2x2 ring of experimentation-phase boxes inside a dashed loop
%      enclosure; white shared-infrastructure inputs (memory context,
%      phase-filtered toolset) feed the ring from above; numbered step
%      badges replace edge labels; the single accent edge is the dashed
%      early-exit back edge (observation already closes the open question).

% from the build cwd, so try the language-build path (cwd report/<lang>/),
% then the repo-root path (cwd report/, used by the preview renderer), then
% the same-directory fallback (standalone wrappers compiled beside the file).

  {\input{../shared/figures/tikz/_style.tex}}%
  {\IfFileExists{shared/figures/tikz/_style.tex}%

\begin{tikzpicture}[aricanvas]

  % --- the four ReAct stages: a ring owned by the experimentation phase ---
  \node[aribox exp]                            (think) {\figlabel{react.think}};
  \node[aribox exp, right=of think]            (act)   {\figlabel{react.act}};
  \node[aribox exp, below=1.15cm of act]       (obs)   {\figlabel{react.observe}};
  \node[aribox exp, below=1.15cm of think]     (refl)  {\figlabel{react.reflect}};

  % numbered step badges on the outer corners (flow order, no edge labels)
  \node[aribadge] at (think.north west) {1};
  \node[aribadge] at (act.north east)   {2};
  \node[aribadge] at (obs.south east)   {3};
  \node[aribadge] at (refl.south west)  {4};

  % dashed enclosure marks the per-node iteration loop
  \node[ariloop, fit=(think)(act)(obs)(refl)] (loop) {};

  % --- shared infrastructure feeding the loop (white = outside any phase) ---
  \node[aribox, above=0.85cm of think]         (mem)   {\figlabel{F4.memcontext}};
  \node[aribox, above=0.85cm of act]           (tools) {\figlabel{F4.toolset}};

  % --- main cycle (solid ink) ---
  \draw[ariarrow] (think) -- (act);
  \draw[ariarrow] (act)   -- (obs);
  \draw[ariarrow] (obs)   -- (refl);
  \draw[ariarrow] (refl)  -- (think);

  % the one accent edge: dashed early-exit back edge across the ring
  \draw[ariarrow accent, dashed] (obs.north west) to[bend left=12] (think.south east);

  % secondary inputs into the loop
  \draw[ariarrow muted] (mem)   -- (think);
  \draw[ariarrow muted] (tools) -- (act);

  % loop budget note, with the iteration glyph beside the enclosure
  \node[arinote, below=0.3cm of loop] (budget) {\figlabel{F4.maxsteps}};
  \pic at ([xshift=-3.2mm]budget.west) {ariglyph loop};

\end{tikzpicture}
