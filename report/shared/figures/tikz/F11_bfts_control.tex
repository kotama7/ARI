% F11: BFTS control flow — house design language (ari* styles, see _style.tex).
% One BFTS step, left to right: (1) frontier -> (2) LLM-driven selection ->
% (3) parse gate -> (4) LLM-driven expansion -> new child node -> (5) prune
% gate. Solid ink edges are control flow; dashed gray edges are context
% inputs: the two prompt files above the LLM stations, and below the spine
% the deterministic fallback (fallback score incl. diversity bonus, taken
% when the parse gate rejects the LLM index), the v0.9.0 lineage-chaining
% hints (coverage hint into expansion-target selection on the parent side,
% lineage hint into the new child on the child side), and the stagnation
% detector feeding the prune gate. The single vermilion back edge appends
% the child to the frontier; the dotted margin path is the continue loop
% back to the next iteration.
% style.tikzstyles is loaded by shared/preamble.tex / the standalone wrapper;
% the ari* house styles load from _style.tex (path resolved from build cwd:
% language builds run in <lang>/, figure previews run in the report root).
\InputIfFileExists{../shared/figures/tikz/_style.tex}{}{%
  \InputIfFileExists{shared/figures/tikz/_style.tex}{}{%
    \InputIfFileExists{_style.tex}{}{%
      \errmessage{F11: shared/figures/tikz/_style.tex not found}}}}%
\begin{tikzpicture}[aricanvas]

  % --- spine: one BFTS step, five numbered stations plus the child ------
  \node[aribox]                                    (front) {\figlabel{F11.frontier}};
  \node[ariboxwide, fill=phaseexp, right=of front] (sel)   {\figlabel{F11.select}};
  \node[aribox gate, right=of sel]                 (dec)   {\figlabel{F11.parse}};
  \node[aribox exp, right=of dec]                  (exp)   {\figlabel{F11.expand}};
  \node[aribox, right=of exp]                      (child) {\figlabel{F11.child}};
  \node[aribox gate, right=of child]               (prune) {\figlabel{F11.prune}};

  % --- numbered step badges (PaperBench grammar; no edge-order labels) --
  \node[aribadge] at (front.north west) {1};
  \node[aribadge] at (sel.north west)   {2};
  \node[aribadge] at (dec.north west)   {3};
  \node[aribadge] at (exp.north west)   {4};
  \node[aribadge] at (prune.north west) {5};

  % --- prompt files above the two LLM-driven stations -------------------
  \node[aribox, above=0.7cm of sel] (psel) {\figlabel{F11.prompt.select}};
  \node[aribox, above=0.7cm of exp] (pexp) {\figlabel{F11.prompt.expand}};

  % --- context row below the spine ---------------------------------------
  \node[ariboxwide, below=0.7cm of dec] (fb)
        {\figlabel{F11.fallback}\\+ \figlabel{F11.diversity}};
  \node[aribox, below=0.7cm of exp]   (cov)  {\figlabel{F11.hint.coverage}};
  \node[aribox, below=0.7cm of child] (lin)  {\figlabel{F11.hint.lineage}};
  \node[aribox, below=0.7cm of prune] (stag) {\figlabel{F11.stagnation}};

  % --- primary control flow ----------------------------------------------
  \draw[ariarrow] (front) -- (sel);
  \draw[ariarrow] (sel)   -- (dec);
  \draw[ariarrow] (dec)   -- (exp)
        node[arinote, midway, above=1.5pt] {\figlabel{F11.ok}};
  \draw[ariarrow] (exp)   -- (child);
  \draw[ariarrow] (child) -- (prune);

  % --- deterministic fallback branch (parse failure) ---------------------
  \draw[ariarrow] (dec.south) -- (fb.north)
        node[arinote, midway, right=2pt] {\figlabel{F11.fail}};
  \draw[arielbow vh] (fb.east) to ([yshift=-2.5mm]exp.west);

  % --- context inputs (dashed): prompts, lineage-chaining hints, stagnation
  \draw[ariarrow muted] (psel) -- (sel);
  \draw[ariarrow muted] (pexp) -- (exp);
  \draw[ariarrow muted] (cov)  -- (exp);
  \draw[ariarrow muted] (lin)  -- (child);
  \draw[ariarrow muted] (stag) -- (prune);

  % --- back edges: append (the one accent edge) over the top, continue
  %     along the bottom margin to the next iteration ----------------------
  \draw[ariarrow accent, rounded corners=7pt]
        (child.north) -- ([yshift=20mm]child.north) -| (front.north);
  \node[arinote] at ([yshift=20mm]dec.north) {\figlabel{F11.append}};
  \draw[arifeedback]
        (prune.east) -- ([xshift=7mm]prune.east)
        |- ([yshift=-8mm]lin.south) -| (front.south);
  \node[arinote] at ([yshift=-8mm]cov.south) {\figlabel{F11.continue}};
\end{tikzpicture}
