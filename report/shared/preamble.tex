% Strict line-breaking. We deliberately do NOT use \sloppy globally:
% \sloppy with a large \emergencystretch suppresses Overfull warnings
% AND silently lets text protrude into the margin. We want TeX to keep
% reporting overfull boxes so we can fix the wording / use \code{}
% wrapping. A small \emergencystretch is fine — it just gives the
% line-breaker a little room to find a clean break on the last pass.
\setlength{\emergencystretch}{1em}
% \code{} renders an identifier in monospace with break-anywhere splits
% via seqsplit. Declared robust so it survives moving arguments
% (\caption{}, \section{}).
\DeclareRobustCommand{\code}[1]{\texttt{\seqsplit{#1}}}

% ----------------------------------------------------------------- CJK fonts
% Loaded only when the engine is lualatex/xelatex AND a CJK-capable font set
% can be resolved. If the host has no Noto CJK installed, the build still
% completes but ja/zh glyphs render as missing-character warnings.
% luatexja-fontspec is only loaded when the document declares itself as
% needing CJK support. Set \AriCJK in the language-specific main.tex
% (ja/main.tex, zh/main.tex) to opt in. Loading it for en/main.tex
% causes a luatexja-microtype patch warning we don't need.
% Engine + language-aware CJK setup. Set \AriCJK in the doc's main.tex
% to opt in; \AriLangJa / \AriLangZh select which lang's fonts to load.
% This lets `make4ht -ul` drive lualatex even for zh (HTML build),
% while keeping the canonical zh PDF on xelatex+ctex.
\ifdefined\luatexversion
  \ifdefined\AriCJK
    \ifdefined\AriLangJa
      \IfFileExists{luatexja-fontspec.sty}{%
        \PassOptionsToPackage{nopatch=microtype}{luatexja}%
        \usepackage{luatexja-fontspec}%
        \IfFontExistsTF{NotoSerifCJKjp-Regular}{%
          \setmainjfont{NotoSerifCJKjp-Regular}[BoldFont=NotoSansCJKjp-Bold]%
          \setsansjfont{NotoSansCJKjp-Regular}%
        }{}%
      }{}%
    \fi
    \ifdefined\AriLangZh
      \usepackage{fontspec}%
      \IfFontExistsTF{FandolSong-Regular}{%
        \setmainfont{FandolSong-Regular}[BoldFont=FandolSong-Bold]%
      }{}%
    \fi
  \fi
\fi
\ifdefined\XeTeXversion
  \ifdefined\AriCJK
    \ifdefined\AriLangJa
      % XeLaTeX + Japanese: xeCJK + HaranoAji (Adobe-Japan1 CIDs, ships
      % with TinyTeX). Without this branch the engine substitutes
      % missing glyphs with the default ``.notdef'' rectangle, which
      % rasterizes to an F-shaped tofu and pollutes the whole document.
      \IfFileExists{xeCJK.sty}{%
        \usepackage{xeCJK}%
        \IfFontExistsTF{HaranoAjiMincho-Regular.otf}{%
          \setCJKmainfont{HaranoAjiMincho-Regular.otf}[%
            BoldFont=HaranoAjiMincho-Bold.otf,%
            ItalicFont=HaranoAjiMincho-Regular.otf]%
          \setCJKsansfont{HaranoAjiGothic-Regular.otf}[%
            BoldFont=HaranoAjiGothic-Bold.otf]%
        }{}%
      }{}%
    \fi
    \ifdefined\AriLangZh
      \IfFileExists{ctex.sty}{%
        \usepackage[fontset=fandol]{ctex}%
      }{}%
    \fi
  \fi
\fi

% ----------------------------------------------------------------- Unicode-aware monospace (lualatex/xelatex only)
% The default T1-encoded \texttt / \Verbatim font is `lmtt` which lacks
% several characters that occur verbatim in our snapshot prompts (U+2014
% em-dash, U+2013 en-dash, U+2192 →). Under lualatex / xelatex we have
% fontspec available; force the mono font to the OpenType Latin Modern Mono
% which carries the full Unicode range. This eliminates "Missing character"
% log noise from the prompts appendix without changing the snapshot bytes.
\ifdefined\directlua
  \usepackage{fontspec}
  \IfFontExistsTF{Latin Modern Mono}{%
    \setmonofont{Latin Modern Mono}[AutoFakeBold=2]%
  }{}
\else\ifdefined\XeTeXversion
  \usepackage{fontspec}
  \IfFontExistsTF{Latin Modern Mono}{%
    \setmonofont{Latin Modern Mono}[AutoFakeBold=2]%
  }{}
\fi\fi

% ----------------------------------------------------------------- verbatim (for Appendix prompt listings)
% fvextra (not base fancyvrb) is what implements real line wrapping:
% `breaklines`, `breakanywhere`, `breaksymbol`. Loading only fancyvrb
% silently ignores those keys and the long prompt lines run off the page.
\usepackage{fvextra}
% \PromptListing{<file>}{<caption>}{<label>}: verbatim-include a snapshot
% prompt from shared/appendix/prompts/. The first 3 lines (snapshot header)
% are skipped — the reader sees only the prompt text. breakanywhere lets a
% line wrap even inside a long unspaced token (e.g. a 500-char JSON line).
\fvset{fontsize=\footnotesize, frame=single, framesep=4pt,
       breaklines=true, breakanywhere=true, breaksymbolleft={},
       breaksymbolright={}, numbers=none}
\newcommand{\PromptListing}[3]{%
  \subsection{#2}\label{#3}%
  \VerbatimInput[firstline=4]{../shared/appendix/prompts/#1}%
}
